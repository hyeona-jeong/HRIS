CREATE TABLE QA_COMMENT(
  idx INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fo_idx INT NOT NULL,
  writer VARCHAR(8) NOT NULL,
  submit_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
  edit_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL,
  FOREIGN KEY (`fo_idx`) REFERENCES `Q_A` (`idx`),
  FOREIGN KEY (`writer`) REFERENCES `MAIN_TABLE` (`NAME_KOR`)
 );
